| tstats summariesonly=true allow_old_summaries=true count as task_events
    FROM datamodel=Endpoint.Processes
    WHERE Processes.process_name="schtasks.exe"
        AND Processes.process="*/create*"
    BY Processes.dest, Processes.user, bin(_time, 1h)
| rename Processes.* as *
| stats avg(task_events) as avg_hr stdev(task_events) as stdev_hr
        latest(task_events) as current_hr by dest, user
| eval upper_bound=round(avg_hr + (2.5 * stdev_hr), 2)
| where current_hr > upper_bound AND current_hr > 3
| eval anomaly_score=round((current_hr - avg_hr) / max(stdev_hr, 0.1), 2)
| eval alert="STATISTICAL ANOMALY - Unusual task creation volume"
| eval technique="T1053.005"
| table dest, user, current_hr, avg_hr, upper_bound, anomaly_score, alert, technique
| sort -anomaly_score

| tstats summariesonly=true allow_old_summaries=true count min(_time) as task_time
    FROM datamodel=Endpoint.Processes
    WHERE Processes.process_name="schtasks.exe"
        AND Processes.process="*/create*"
    BY Processes.dest, Processes.user, Processes.process
| rename Processes.* as *
| rename dest as join_host
| join type=inner join_host [
    | tstats summariesonly=true allow_old_summaries=true count min(_time) as net_time
        FROM datamodel=Network_Traffic.All_Traffic
        WHERE All_Traffic.action="allowed"
            AND All_Traffic.dest_port IN (80, 443, 8080, 8443, 4444, 1337, 53, 636, 6667)
            AND All_Traffic.app!="ssl"
        BY All_Traffic.src, All_Traffic.dest_ip, All_Traffic.dest_port, All_Traffic.app
    | rename All_Traffic.src as join_host
    | rename All_Traffic.* as *]
| eval time_delta_min=round(abs(task_time - net_time) / 60, 1)
| where time_delta_min <= 15
| eval risk=case(
    dest_port IN (4444, 1337, 6667), "CRITICAL - Non-standard C2 port",
    dest_port IN (8080, 8443),       "HIGH - Alternate web port",
    dest_port=53,                    "HIGH - DNS C2 possible",
    true(),                          "MEDIUM - Standard port, review")
| eval technique="T1053.005", tactic="Command and Control"
| table task_time, join_host, user, dest_ip, dest_port, app, time_delta_min, risk, technique
| sort -risk, time_delta_min
